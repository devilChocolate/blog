    二叉搜索树

    wangkai 2022/1/10

    # 二叉搜索树的概念

    • 1. 二叉搜索树是一种特殊的二叉树
    • 2. 若左子结点不为空,则左子结点的值小于其父结点
    • 2. 若右子结点不为空,则右子结点的值大于其父结点

    # 二叉搜索树的操作

    • 1 .查找：递归查找是否存在key

    • 2 .插入：原树中不存在key，插入key返回true，否则返回false

    • 3. 删除某个结点

    • 4.先序遍历

    • 5.中序遍历

    • 6.后序遍历

    • 7.层序遍历

    # 自定义封装一颗二叉搜索树

      /**
  * @description 二叉搜索树节点类
  * @param {*} key 节点key
  * @param {*} value 节点value
  * @Date 2021/11/25
  * @author wangkai
  * @return
  */
  //查找插入节点
  const insertFn = Symbol("insertFn");
  //先序遍历
  const preorderTraversalFn = Symbol("preorderTraversalFn");
  //中序遍历
  const mediumOrderTraversalFn = Symbol("mediumOrderTraversalFn");
  //后序遍历
  const postorderTraversalFn = Symbol("postorderTraversalFn");
  //删除节点
  const deleteFn = Symbol("deleteFn");
  //删除叶子节点
  const deleteLeafNode = Symbol("deleteLeafNode");
  // 删除只有左子或右子的节点
  const deleteOneChildren = Symbol("deleteOneChildren")
  // 删除有两个子节点的节点
  const deleteFullNode = Symbol('deleteFullNode')
  //根据key查找节点
  const searchFn = Symbol('searchFn')
  //查找前驱节点
  const searchPrecursorFn = Symbol('searchPrecursorFn')
  //查找后继节点
  const searchSuccessorFn = Symbol('searchSuccessorFn')

  //节点类
  class Node {
    constructor(key, value) {
      //节点的key
      this.key = key;
      //节点的value
      this.value = value;
      //左子节点索引
      this.leftNode = null;
      //右子节点索引
     this.rightNode = null;
    }
  }

  /**
  * @description 二叉搜索树类
  * @return
  */
  class BinarySearchTree {
    constructor() {
      //根节点
      this.root = null;
    }

    /**
    * @description 添加节点
    * @param {*} key 新增节点key
    * @param {*} value 新增节点value
    * @return
    */
    insert(key, value) {
      const newNode = new Node(key, value);
      if (!this.root) {
        //当前是一颗空树，新增节点为根节点
        this.root = newNode;
        return;
      } else {
        //查找:新增节点的位置
        let selectNode = this.root;
        this[insertFn](selectNode, newNode);
      }
    }

    /**
    * @param {*} 删除节点key
    * @return
   */
    delete(key) {
      let prev = null //保存被删除节点的父节点
      let current = this.root
      while (current.key !== key) {
        prev = current
        if(key < current.key){
          current = current.leftNode
        }else{
          current = current.rightNode
        }
        if(!current){
          //未找到要删除到节点
          return false
        }
      }
      if(!current.leftNode && !current.rightNode){
         //被删除节点是叶子节点
         return this[deleteLeafNode](prev,current)
       }else if(current.leftNode && current.rightNode){
         //被删除的节点度为2（存在两个子节点）
         return this[deleteFullNode](prev,current)
       }else{
         //如果存在一个子节点 左节点或右节点
         return this[deleteOneChildren](prev,current)
       }
    }

    //先序遍历
    preorderTraversal(hander) {
      if (!this.root) {
        return;
      }
      this[preorderTraversalFn](this.root, hander);
    }


    //中序遍历
    mediumOrderTraversal(hander) {
      if (!this.root) {
        return;
      }
      this[mediumOrderTraversalFn](this.root, hander);
    }


    //后序遍历
    postorderTraversal(hander) {
      if (!this.root) {
        return;
      }
      this[postorderTraversalFn](this.root, hander);
    }


    /**
    * @description 查找节点
    * @param {*} key
    */
    search(key){
      const node = this[searchFn](key)
        return node ? {key: key,value:node.value} : null
    }


    /**
    * @description 查找最大值
    * @return maxNode
    */
    max() {
      if (this.root) {
        let node = this.root;
        while (node.rightNode) {
          node = node.rightNode;
        }
       return { key: node?.key, value: node?.value };
      }
      return null;
    }


    /**
    * @description 查找最小值
    * @return minNode
    */
    min() {
      if (this.root) {
        let node = this.root;
          while (node.leftNode) {
          node = node.leftNode;
        }
        return { key: node?.key, value: node?.value };
      }
      return null;
    }


    /**
    * @description 删除叶子节点方法
    * @param {*} prev deleteNode 父节点
    * @param {*} deleteNode 被删除节点
    */
    [deleteLeafNode](prev,deleteNode){
      //当前节点是叶子节点
      if(this.root === deleteNode){
        //当前节点未根节点
        this.root = null
        return true
      }else{
        if(deleteNode.key < prev.key){
          //左叶节点
          prev.leftNode = null
          return true
        }else{
           //左叶节点
           prev.rightNode = null
           return true
        }
      }
    }

    /**
    * @description 删除有左子或只有右子的节点
    * @param {*} prev 被删除节点的父节点
    * @param {*} deleteNode 被删除的节点
    * @returns
    */
    [deleteOneChildren](prev,deleteNode){
      if(this.root === deleteNode){
        //如果是根节点
        if(deleteNode.leftNode){
          this.root = deleteNode.leftNode
          return true
        }else{
          this.root = deleteNode.rightNode
          return true
        }
      }else{
        if(deleteNode.leftNode){
          if(prev.key > deleteNode.key){
            prev.leftNode = deleteNode.leftNode
            return true
           }else{
             prev.rightNode = deleteNode.leftNode
             return true
           }
        }else{
          if(prev.key > deleteNode.key){
            prev.leftNode = deleteNode.rightNode
            return true
          }else{
            prev.rightNode = deleteNode.rightNode
            return true
          }
        }
      }
    }
  
    /**
    * @description 删除有两个子节点的节点
    * @param {*} prev 被删除节点的父节点
    * @param {*} deleteNode 被删除的节点
    * @returns
    */
    [deleteFullNode](prev, deleteNode) {
      //如果存在两个叶子节点
      const {prevNode,precursorNode} = this[searchPrecursorFn](deleteNode); //前驱节点 前驱节点的父节点
      let success = false
      if (this.root === deleteNode) {
        precursorNode.leftNode = this.root.leftNode
        precursorNode.rightNode = this.root.rightNode
        this.root = precursorNode
        prevNode.rightNode = null
        success = true
      } else {
          if (prev.key > deleteNode.key) {
            prevNode.leftNode = precursorNode.leftNode
            prev.leftNode = precursorNode
            precursorNode.leftNode = deleteNode.leftNode
            precursorNode.rightNode = deleteNode.rightNode
            success = true
         } else {
            prevNode.leftNode = precursorNode.rightNode
           prev.rightNode = precursorNode
           precursorNode.leftNode = deleteNode.leftNode
           precursorNode.rightNode = deleteNode.rightNode
           success = true
        }
      }
      return success
    }

    /**
    * @description 查找node的前驱节点
    * @param {*}} node
    */
    [searchPrecursorFn](node) {
      let prevNode = node //前驱节点的父节点
      let precursorNode = node.leftNode;
      while (precursorNode.rightNode) {
        prevNode = precursorNode
        precursorNode = precursorNode.rightNode;
      }
      return {prevNode,precursorNode};
    }


    /**
    * @description 查找node的后继节点
    * @param {*}} node
    */
    [searchSuccessorFn](node){
      let prevNode = node //后继节点的父节点
      let successNode = node.rightNode
      while (successNode.leftNode) {
        prevNode = successNode
        successNode = successNode.leftNode
     }
      return {prevNode,successNode}
    }

    /**
    * @description 先序遍历节点私有方法
    * @param {*} node 当前遍历的方法
    * @param {*} hander 回调函数处理节点
    */
    [preorderTraversalFn](node, hander) {
      if (!!node) {
        hander(node);
        this[preorderTraversalFn](node.leftNode, hander);
        this[preorderTraversalFn](node.rightNode, hander);
      }
    }


    /**
    * @description 中序遍历节点私有方法
    * @param {*} node 当前遍历的方法
    * @param {*} hander 回调函数处理节点
    */
    [mediumOrderTraversalFn](node, hander) {
      if (!!node) {
        this[mediumOrderTraversalFn](node.leftNode, hander);
        hander(node);
        this[mediumOrderTraversalFn](node.rightNode, hander);
      }
    }


    /**
    * @description 后序遍历节点私有方法
    * @param {*} node 当前遍历的方法
    * @param {*} hander 回调函数处理节点
    */
    [postorderTraversalFn](node, hander) {
      if (!!node) {
        this[postorderTraversalFn](node.leftNode, hander);
        this[postorderTraversalFn](node.rightNode, hander);
        hander(node);
      }
    }


    /**
    * @description 查找并且插入节点私有方法
    * @param {*} node 当前节点
    * @param {*} newNode 新插入的节点
    */
    [insertFn](node, newNode) {
      if (newNode.key < node.key) {
        if (!node.leftNode) {
          node.leftNode = newNode;
          return;
        } else {
          this[insertFn](node.leftNode, newNode);
        }
      }else if (newNode.key > node.key) {
        if (!node.rightNode) {
          node.rightNode = newNode;
          return;
        } else {
          this[insertFn](node.rightNode, newNode);
        }
      } else {
        //等于 存在相同key 根据实际业务,替换value值/抛出异常等 此处使用抛出异常，禁止两个相同的key
       throw new Error("存在相同的key,错误！！！");
      }
    }


    /**
    * @description 私有方法 根据key查找node节点
    * @param {*} key
    * @returns
    */
    [searchFn](key){
      let current = this.root
        while (current.key !== key) {
          if(key < current.key){
            current = current.leftNode
          }else{
            current = current.rightNode
          }
          if(!current){
            return null
          }
        }
      return current
    }

    /**
    * @description 删除操作查找节点
    * @return deleteNode
    */
    [deleteFn](key) {
      const node = this.search(key)
    }
  }
