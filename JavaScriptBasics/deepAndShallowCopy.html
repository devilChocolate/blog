    # 什么是浅拷贝

    • 开辟一个新的空间,将源数据拷贝到该对象中。如果源对象的属性是基本数据类型则拷贝它的值,如果是引用类型则拷贝它的地址值(引用)
    • ‘=’号赋值并不是浅拷贝,浅拷贝指如果数据第一层是基本数据类型则拷贝值,如果是引用类型就拷贝其地址(引用值),例如Array的concat和slice方法
      //非浅拷贝  
  let a1 = "张三"
  let a2 = a1
  a1 = "李四"
  
  //非浅拷贝
  const a1 = {name: '张三'}
  const a2 = a1
  a2.name = '李四'
------------------------------------------------------
  //浅拷贝
  const arr1 = [1,2,3,{name: '张三'}]
  const arr2 = arr1.slice()
  const arr3 = arr2.concat()

    # 自定义浅拷贝方法

      function shallowCopy(context) {
    const result = Array.isArray(context) ? [] : {}
    if(typeof context === 'object'){
      for(let key in context){
        if(context.hasOwnProperty(key)){
          //是否是本身的属性 而不是原型链上的属性
          //如果是基本类型的值拷贝值 如果是引用类型的拷贝其存储的地址值
          result[key] = context[key]
        }
      }
    }else{
      result = context
    }
    return result
  }
  const o1 = {
    name: 'zhangsan',
    live:['篮球','足球']
  }
  const o2 = shallowCopy(o1)
  o2.name = "李四"
  o2.live.push("哈哈哈")
  //因为name保存的基本路线数据,所以浅拷贝会拷贝其值,改变其中一个不会影响其他数据,而live是保存了地址值,改变其中一个会使另一个发生变化
  console.log(o1); //输出 { name: 'zhangsan', live: [ '篮球', '足球', '哈哈哈' ] }
  console.log(o2); //输出 { name: '李四', live: [ '篮球', '足球', '哈哈哈' ] }

    # 什么是深拷贝

    • 当一个数据复制给另一个数据后,如果其中一个数据发生了改变,另一个不会发生变化
    
  let obj1 = {name: "张三",age:18}
  let obj2 = obj1
  obj1.name = "李四"
  console.log(obj2); // 输出 { name: '李四', age: 18 }

    # 深浅拷贝的原理

    • 造成深拷贝或浅拷贝的原因是因为不同的数据类型(基本类型和引用类型)在内存中的存储位置不同。基本类型数据的变量及其数据都在栈内存中保存,引用类型数据的变量和地址(数据在堆内存空间中分配的内存地址)保存在栈内存中,其数据保存在堆内存当中。当进行赋值操作时(a = b),将原变量在栈内存中保存的内容复制给了新的变量。如果是基本类型的数据,就会为新变量重新分配一个空间,保存变量名和数据；如果是引用数据类型,就会为新变量重新分配一个空间保存变量名和地址。由于引用类型数据是保存了相同的地址,所以共同指向在堆内存当中的数据。因此引用类型的赋值是浅拷贝而基本类型数据的赋值是深拷贝

    # Array的slice和concat方法

    • 如果数组只有一层,slice和concat方法是深拷贝
      const a = [1,2,3]
  const b = a.slice()
  const c = a.concat()
  a[1] = 100
  console.log(b); //输出 [ 1, 2, 3 ]
  console.log(c); //输出 [ 1, 2, 3 ]
    • 如果数组具有多层,slice和concat方法是浅拷贝
      const a = [[1,2,3]]
  const b = a.slice()
  const c = a.concat()
  a[0][0] = 100
  console.log(b); //输出 [ [ 100, 2, 3 ] ]
  console.log(c); //输出 [ [ 100, 2, 3 ] ]

    # 引用类型实现深拷贝的方式

    • 1. 使用递归实现深拷贝
      function deepClone(obj) {
    let result = Array.isArray(obj) ? [] : {};
    if (typeof obj === "object") {
      //如果是引用类型
      for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
          //不复制原型链的内容
          if (obj[key] && typeof obj[key] === "object") {
            //obj是不是对象类型
            result[key] = deepClone(obj[key]);
          } else {
            result[key] = obj[key];
          }
        }
      }
    } else {
      //基本类型
      result = obj;
    }
    return result;
  }
  const obj1 = {
    name: "zhangsan",
    age: 19,
    info: {
      address: "安徽省",
      live: {
        aa: "篮球",
        bb: "羽毛球",
      },
      cuss: [1, 2, 3, 4],
    },
  };
  const obj2 = deepClone(obj1);
  obj2.name = "我是深拷贝复制的对象";
  obj2.info.cuss = [9, 8, 7, 6];
  console.log(obj1); // 输出 {name: 'zhangsan',age: 19,info: {address: '安徽省', live: { aa: '篮球', bb: '羽毛球' },cuss: [ 1, 2, 3, 4 ]}}
  console.log(obj2); //输出 {name: '我是深拷贝复制的对象',age: 19,info: {address: '安徽省',live: { aa: '篮球', bb: '羽毛球' },cuss: [ 9, 8, 7, 6 ]}}
